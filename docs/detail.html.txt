    <script>
        const classes = ["Lớp 5A", "Lớp 3B"];
        let currentClass = 0;

        const students = [
        {
        name: "Hoàng Anh",
        avatar: "https://i.pravatar.cc/40?1",
        learn: "✔️",
        nghe: "correct",
        doc: "wrong",
        viet: "pending",
        noi: "wrong"
        },
        {
        name: "Công Vinh",
        avatar: "https://i.pravatar.cc/40?2",
        learn: "⚠️",
        nghe: "wrong",
        doc: "wrong",
        viet: "pending",
        noi: "wrong"
        }
        ];

        // render tabs
        function renderTabs() {
        let html = "";
        classes.forEach((c, i) => {
        html += `<div class="tab ${i===currentClass?'active':''}" data-i="${i}">${c}</div>`;
        });
        $("#tabs").html(html);
        }

        // icon
        function getIcon(status) {
        if (status === "correct") return '<span class="status correct">✔</span>';
        if (status === "wrong") return '<span class="status wrong">✖</span>';
        return '<span class="status pending">📘</span>';
        }

        // render table
        function renderTable() {
        let html = "";
        students.forEach(s => {
        html += `
        <tr>
        <td class="d-flex align-items-center gap-2">
        <img src="${s.avatar}" class="avatar">
        ${s.name}
        </td>
        <td>${s.learn}</td>
        <td>${getIcon(s.nghe)}</td>
        <td>${getIcon(s.doc)}</td>
        <td>${getIcon(s.viet)}</td>
        <td>${getIcon(s.noi)}</td>
        </tr>
        `;
        });
        $("#studentTable").html(html);
        }

        // click tab
        $(document).on("click", ".tab", function () {
        currentClass = $(this).data("i");
        renderTabs();
        renderTable();
        });

        // init
        renderTabs();
        renderTable();
    </script>
