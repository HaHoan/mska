    <script>
        const questions = [
        { q: "Chọn hình đúng 1", answers: ["A","B","C"] },
        { q: "Chọn hình đúng 2", answers: ["A","B","C"] },
        { q: "Chọn hình đúng 3", answers: ["A","B","C"] },
        { q: "Chọn hình đúng 4", answers: ["A","B","C"] }
        ];

        let current = 0;
        let answers = {};
        let time = 120;

        // render nav
        function renderNav() {
        let html = "";
        questions.forEach((q, i) => {
        let cls = "question-item";
        if (i === current) cls += " active";
        if (answers[i]) cls += " answered";

        html += `<div class="${cls}" data-i="${i}">${i+1}</div>`;
        });
        $("#questionNav").html(html);
        }

        // render question
        function renderQuestion() {
        let q = questions[current];
        $("#questionText").text(q.q);

        let html = "";
        q.answers.forEach(a => {
        let selected = answers[current] === a ? "selected" : "";
        html += `
        <div class="col-4 option ${selected}" data-value="${a}">
        <img src="https://picsum.photos/200?random=${Math.random()}">
        <div>${a}</div>
        </div>`;
        });

        $("#options").html(html);
        }

        // chọn đáp án
        $(document).on("click", ".option", function () {
        $(".option").removeClass("selected");
        $(this).addClass("selected");

        answers[current] = $(this).data("value");
        updateProgress();
        renderNav();
        });

        // chuyển câu
        $(document).on("click", ".question-item", function () {
        current = $(this).data("i");
        renderNav();
        renderQuestion();
        });

        // progress
        function updateProgress() {
        let percent = Object.keys(answers).length / questions.length * 100;
        $("#progressBar").css("width", percent + "%");
        }

        // timer
        function startTimer() {
        setInterval(() => {
        time--;
        let m = String(Math.floor(time / 60)).padStart(2,'0');
        let s = String(time % 60).padStart(2,'0');
        $("#timer").text(`${m}:${s}`);

        if (time <= 0) {
        alert("Hết giờ!");
        }
        }, 1000);
        }

        // submit
        $("#submitBtn").click(function () {
        alert("Bạn đã nộp bài!");
        });

        // init
        renderNav();
        renderQuestion();
        startTimer();
    </script>
